# Auction Hub — C4 Architecture Diagrams

## Level 1: System Context

```mermaid
C4Context
    title System Context — Auction Hub

    System(auctionHub, "Auction Hub", "Platform for creating auctions and running real-time bidding")
    
    Person(owner, "Auction Owner", "Creates auctions, manages lots, runs bidding rooms")
    Person(bidder, "Bidder", "Joins rooms, places bids on lots")

    System_Ext(email, "Email Server", "Sends invite and confirmation emails to bidders")
    System_Ext(s3, "S3 / MinIO", "Stores lot images")
    System_Ext(google, "Google Identity", "OAuth 2.0 / Google Sign-In")
    System_Ext(sentry, "Sentry", "Error tracking and telemetry")

    Rel(owner, auctionHub, "Manages auctions, lots, rooms", "HTTPS / WS")
    Rel(bidder, auctionHub, "Registers, confirms invite, places bids", "HTTPS / WS")
    Rel(auctionHub, email, "Sends invite / confirmation emails", "SMTP")
    Rel(auctionHub, s3, "Uploads / serves lot images", "S3 API")
    Rel(auctionHub, google, "Verifies ID tokens (Sign in with Google)", "OAuth 2.0")
    Rel(auctionHub, sentry, "Reports errors and performance data", "HTTPS")
```

## Level 2: Container

```mermaid
C4Container
    title Container Diagram — Auction Hub (Refined)

    Person(owner, "Auction Owner", "Creates and manages auctions")
    Person(bidder, "Bidder", "Joins auctions and places bids")

    System_Boundary(hub, "Auction Hub") {
        Container(browser, "Browser Client", "React (Next.js UI)", "CRM interface for owners and bidding UI for participants")

        Container(nginx, "Nginx", "Reverse Proxy", "Routes traffic to client and API (production)")

        Container(bff, "Next.js BFF", "Next.js Server Runtime", "Handles SSR, session management, and acts as API proxy")
        
        Container(api, "NestJS API", "Node.js, NestJS", "Provides REST endpoints and WebSocket gateway for real-time auction logic")
        
        ContainerDb(pg, "PostgreSQL", "PostgreSQL 16", "Persistent data: users, auctions, lots, bids history, results")
        
        ContainerDb(redis, "Redis", "Redis Stack", "Runtime state: active rooms, bids, sessions, pub/sub events")
    }

    System_Ext(email, "Email Server", "Sends invitations and confirmations")
    System_Ext(s3, "S3 / MinIO", "Stores uploaded images")
    System_Ext(google, "Google Identity", "OAuth 2.0 / Google Sign-In")
    System_Ext(sentry, "Sentry", "Error tracking and telemetry")

    Rel(owner, browser, "Uses", "HTTPS")
    Rel(bidder, browser, "Uses", "HTTPS")
    
    Rel(browser, nginx, "Requests pages / API / WS", "HTTPS")
    Rel(nginx, bff, "Proxies page and /api requests", "HTTP")
    Rel(nginx, api, "Proxies WebSocket connections", "WS")
    Rel(bff, api, "Calls API", "HTTP")
    
    Rel(bff, redis, "Reads/writes sessions", "TCP")
    
    Rel(api, pg, "Reads/writes data", "SQL")
    Rel(api, redis, "Manages runtime state, pub/sub", "TCP")
    
    Rel(api, email, "Sends emails", "SMTP")
    Rel(api, s3, "Generates presigned URLs", "S3 API")
    Rel(api, google, "Verifies Google ID tokens", "OAuth 2.0")

    Rel(browser, google, "Sign in with Google (GIS)", "HTTPS")
    Rel(browser, s3, "Uploads files via presigned URL", "HTTPS")

    Rel(browser, sentry, "Client errors / telemetry", "HTTPS")
    Rel(api, sentry, "Server errors / telemetry", "HTTPS")
```

## Level 3: Component — NestJS API

```mermaid
C4Component
    title Component Diagram — NestJS API (Room Engine)

    Container_Boundary(api, "NestJS API") {
        Component(httpApi, "HTTP API Layer", "Controllers", "Auth, Auctions, Lots, Buyers CRUD endpoints")
        
        Component(googleAuth, "GoogleAuthService", "Service", "Verifies Google ID tokens, links Google accounts to users")
        
        Component(wsGateway, "WebSocket Gateway", "Gateway", "Handles real-time connections and events (join, placeBid, control actions)")
        
        Component(roomEngine, "RoomEngine", "Core", "Runtime state machine: bidding, lot transitions, room lifecycle control")
        
        Component(redisRepo, "RedisStateRepository", "Repository", "Stores room state: active lot, bids, participants, TTL")
        
        Component(persistence, "Persistence Layer", "Services", "Users, Auctions, Lots, BidHistory (PostgreSQL)")
        
        Component(infra, "Infrastructure Services", "Services", "Email delivery, S3 storage (presigned URLs)")
    }

    ContainerDb(redis, "Redis", "Runtime state storage")
    ContainerDb(pg, "PostgreSQL", "Persistent data storage")

    System_Ext(email, "Email Server", "SMTP")
    System_Ext(s3, "S3 / MinIO", "Object storage")
    System_Ext(google, "Google Identity", "OAuth 2.0")

    %% HTTP
    Rel(httpApi, persistence, "CRUD operations", "HTTP")

    %% Google auth
    Rel(httpApi, googleAuth, "POST /auth/google", "")
    Rel(googleAuth, google, "Verify ID token", "OAuth 2.0")
    Rel(googleAuth, persistence, "Find / create user", "")
    
    %% WS → Engine
    Rel(wsGateway, roomEngine, "join / placeBid / control", "WebSocket")
    
    %% Engine → State
    Rel(roomEngine, redisRepo, "Read/write room state", "")
    Rel(redisRepo, redis, "Room data, TTL", "TCP")
    
    %% Engine → DB
    Rel(roomEngine, persistence, "Persist results, winners, history", "")
    Rel(persistence, pg, "SQL", "")
    
    %% Engine → WS
    Rel(roomEngine, wsGateway, "Broadcast updates", "WebSocket")
    
    %% Infra
    Rel(roomEngine, infra, "Send emails, manage files", "")
    Rel(infra, email, "SMTP", "")
    Rel(infra, s3, "S3 API", "")
```

## Level 3: Component — Next.js Client

```mermaid
C4Component
    title Component Diagram — Next.js Client (Refined)

    Container_Boundary(client, "Next.js App (Browser + BFF)") {
        Component(browserUI, "Browser UI", "React", "CRM pages, auth UI, auction room UI")
        
        Component(bff, "BFF Layer", "Next.js Server", "SSR, session handling, API proxy")
        
        Component(middleware, "Middleware", "Next.js", "Auth checks, redirects")
        
        Component(sessionService, "Session Service", "Service", "Manages sessions in Redis, handles token refresh")
        
        Component(apiServer, "API Server Client", "HTTP Client", "Server-side API calls with JWT")
        
        Component(apiClient, "API Browser Client", "HTTP Client", "Browser calls via BFF (/api proxy)")
        
        Component(roomEngine, "Room Engine", "OOP State Machine", "Admin / Member / Public engines; room state via useSyncExternalStore, Context providers + useXxxRoom() hooks")
        
        Component(socketClient, "WebSocket Client", "Socket.IO", "Handles real-time room communication")
        
        Component(googleAuth, "Google Auth Module", "GIS", "Google Sign-In button, GIS loader, token handoff to API")
        
        Component(uiModules, "UI Modules", "React", "Forms, modals, notifications, tables")
    }

    ContainerDb(redis, "Redis", "Session storage")
    Container(api, "NestJS API", "Backend")
    System_Ext(google, "Google Identity", "OAuth 2.0")

    %% Browser → BFF
    Rel(browserUI, bff, "HTTP (SSR, navigation)")
    
    %% Middleware
    Rel(middleware, sessionService, "Checks session")
    
    %% Session
    Rel(bff, sessionService, "Uses")
    Rel(sessionService, redis, "Read/write session")
    
    %% API calls
    Rel(bff, apiServer, "Delegates")
    Rel(apiServer, api, "HTTP + JWT")
    
    Rel(browserUI, apiClient, "Client actions")
    Rel(apiClient, bff, "HTTP (/api proxy)")
    
    %% Room Engine + WebSocket
    Rel(browserUI, roomEngine, "subscribe() / actions (placeBid, nextLot...)")
    Rel(roomEngine, socketClient, "Socket lifecycle, events")
    Rel(socketClient, api, "WebSocket (/ws/room)")
    
    %% Google Sign-In
    Rel(browserUI, googleAuth, "Sign in with Google")
    Rel(googleAuth, google, "GIS token flow")
    Rel(googleAuth, apiClient, "POST /auth/google")
    
    %% UI
    Rel(browserUI, uiModules, "Uses")
```
