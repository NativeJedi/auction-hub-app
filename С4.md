# Auction Hub — C4 Architecture Diagrams

## Level 1: System Context

```mermaid
C4Context
    title System Context — Auction Hub

    Person(owner, "Auction Owner", "Creates auctions, manages lots, runs bidding rooms")
    Person(bidder, "Bidder", "Joins rooms, places bids on lots")

    System(auctionHub, "Auction Hub", "Platform for creating auctions and running real-time bidding")

    System_Ext(email, "Email Server", "Sends invite confirmations to bidders")
    System_Ext(s3, "S3 / MinIO", "Stores lot images")

    Rel(owner, auctionHub, "Manages auctions, lots, rooms", "HTTPS / WS")
    Rel(bidder, auctionHub, "Registers, confirms invite, places bids", "HTTPS / WS")
    Rel(auctionHub, email, "Sends invite emails", "SMTP")
    Rel(auctionHub, s3, "Uploads / serves lot images", "S3 API")
```

## Level 2: Container

```mermaid
C4Container
    title Container Diagram — Auction Hub

    Person(owner, "Auction Owner")
    Person(bidder, "Bidder")

    System_Boundary(hub, "Auction Hub") {
        Container(client, "Next.js Client", "React, Next.js 15", "CRM for owners, bidding UI for members. SSR + CSR")
        Container(api, "NestJS API", "Node.js, NestJS", "REST API + WebSocket gateway for real-time bidding")
        ContainerDb(pg, "PostgreSQL", "PostgreSQL 16", "Users, auctions, lots, buyers, images metadata")
        ContainerDb(redis, "Redis", "Redis Stack", "Sessions, room state, bids, pub/sub for WS events")
    }

    System_Ext(email, "Email Server", "SMTP")
    System_Ext(s3, "S3 / MinIO", "Object Storage")

    Rel(owner, client, "Uses CRM", "HTTPS")
    Rel(bidder, client, "Joins room, places bids", "HTTPS")
    Rel(client, api, "REST calls (SSR + proxy)", "HTTP")
    Rel(client, api, "Real-time bidding", "WebSocket")
    Rel(client, redis, "Session storage", "TCP")
    Rel(api, pg, "Reads/writes entities", "SQL")
    Rel(api, redis, "Room state, tokens, pub/sub", "TCP")
    Rel(api, email, "Sends invite emails", "SMTP")
    Rel(api, s3, "Presigned URLs, delete", "S3 API")
    Rel(client, s3, "Direct image upload", "HTTPS")
```

## Level 3: Component — NestJS API

```mermaid
C4Component
    title Component Diagram — NestJS API

    Container_Boundary(api, "NestJS API") {
        Component(authCtrl, "AuthController", "Controller", "Register, login, refresh, logout")
        Component(authService, "AuthService", "Service", "Password hashing, token generation")
        Component(tokenService, "TokenService", "Service", "JWT access / refresh / room tokens")

        Component(auctionsCtrl, "AuctionsController", "Controller", "CRUD for auctions")
        Component(auctionsService, "AuctionsService", "Service", "Auction business logic")

        Component(lotsCtrl, "LotsController", "Controller", "CRUD for lots + image management")
        Component(lotsService, "LotsService", "Service", "Lots logic, presigned URLs")

        Component(roomCtrl, "RoomController", "Controller", "Create room, invite, confirm")
        Component(roomGateway, "RoomGateway", "WS Gateway", "join, placeBid, placeLot, finishAuction")
        Component(roomService, "RoomService", "Service", "Room lifecycle, bidding, lot transitions")
        Component(roomRepo, "RoomRepository", "Repository", "Redis-based room state")

        Component(buyersService, "BuyersService", "Service", "Saves winning bidder to DB")
        Component(emailService, "EmailService", "Service", "Sends emails via Nodemailer")
        Component(storageService, "StorageService", "Service", "S3 presigned URLs, delete objects")
        Component(usersService, "UsersService", "Service", "User CRUD")
    }

    ContainerDb(pg, "PostgreSQL")
    ContainerDb(redis, "Redis")
    System_Ext(email, "Email Server")
    System_Ext(s3, "S3 / MinIO")

    Rel(authCtrl, authService, "")
    Rel(authService, tokenService, "")
    Rel(authService, usersService, "")
    Rel(tokenService, redis, "Refresh tokens")

    Rel(auctionsCtrl, auctionsService, "")
    Rel(auctionsService, pg, "TypeORM")

    Rel(lotsCtrl, lotsService, "")
    Rel(lotsService, pg, "TypeORM")
    Rel(lotsService, storageService, "")
    Rel(storageService, s3, "S3 API")

    Rel(roomCtrl, roomService, "")
    Rel(roomGateway, roomService, "")
    Rel(roomService, roomRepo, "")
    Rel(roomRepo, redis, "Room state")
    Rel(roomService, buyersService, "")
    Rel(roomService, emailService, "")
    Rel(buyersService, pg, "TypeORM")
    Rel(emailService, email, "SMTP")
    Rel(roomGateway, redis, "Pub/Sub")
```

## Level 3: Component — Next.js Client

```mermaid
C4Component
    title Component Diagram — Next.js Client

    Container_Boundary(client, "Next.js Client") {
        Component(middleware, "Middleware", "Next.js", "Auth check, redirects")
        Component(apiRoutes, "API Routes", "Route Handlers", "Proxy: auth, auctions, lots, room. Session cookies")
        Component(sessionService, "SessionStorage", "Service", "Redis sessions + auto JWT refresh")

        Component(crmPages, "CRM Pages", "React SSC + Client", "Auctions list, detail, lots management")
        Component(authPage, "Auth Page", "React Client", "Login / Register forms")
        Component(roomAdmin, "Room Admin", "React Client", "Live auction control: lots, bids, participants")
        Component(roomMember, "Room Member", "React Client", "Live bidding UI: place bids, view lot")
        Component(roomInvite, "Room Invite", "React Client", "Invite form + email confirmation")

        Component(apiServer, "API Server Layer", "Axios", "Server-side calls with JWT from session")
        Component(apiClient, "API Client Layer", "Axios", "Browser calls via /api proxy")
        Component(socketClient, "BaseSocket", "Socket.IO", "WebSocket for real-time room events")

        Component(modals, "Modals Module", "React", "createModalRenderer, ConfirmModal")
        Component(forms, "Forms Module", "Zod + RHF", "FormBuilder with field registry")
        Component(notifications, "Notifications", "React Context", "Toast system")
    }

    ContainerDb(redis, "Redis")
    Container(api, "NestJS API")

    Rel(middleware, sessionService, "Checks session")
    Rel(sessionService, redis, "Get/refresh session")
    Rel(apiRoutes, apiServer, "Delegates")
    Rel(apiServer, api, "HTTP + Bearer JWT")
    Rel(apiClient, apiRoutes, "HTTP /api proxy")

    Rel(crmPages, apiServer, "SSR fetch")
    Rel(crmPages, apiClient, "Client mutations")
    Rel(authPage, apiClient, "Login / Register")
    Rel(roomAdmin, socketClient, "WS events")
    Rel(roomMember, socketClient, "WS events")
    Rel(roomInvite, apiClient, "Send/confirm invite")
    Rel(socketClient, api, "WebSocket")
```
